// --- 2. HELPER FUNCTIONS ---

function normalizeSchemeId(name: string, system: string): string {
  const prefix = system.toLowerCase();
  const normalized = name.toLowerCase().trim().replace(/[^a-z0-9]+/g, '_').replace(/^_+|_+$/g, '');
  return ${prefix}_${normalized};
}

// --- 3. EXTRACTION LOGIC ---

function extractSchemes(sourceData: any[], system: "SNAPFIRE" | "SHINOBI"): SchemeKnowledge[] {
  const schemes: SchemeKnowledge[] = [];
  console.log(Processing ${sourceData.length} items for ${system}...);

  for (const play of sourceData) {
    // Determine Name
    const displayName = play.play_name || play.concept?.name || play.coverage_type || 'Unknown Play';
    const internalId = normalizeSchemeId(displayName, system);

    // Extract Routes/Assignments
    const routes: SchemeRoute[] = [];
    
    // Handle Offense (Route Tree)
    if (play.route_tree && Array.isArray(play.route_tree)) {
      play.route_tree.forEach((r: any) => {
        if (r.route && r.route !== 'null') {
          // Clean up position names (e.g., WR1_X -> X)
          const cleanPos = r.position.replace(/WR\d_/, '').replace('_', ' ');
          routes.push({
            receiver: cleanPos,
            route: r.route,
            note: r.assignment
          });
        }
      });
    }
    
    // Handle Defense (Position Assignments)
    if (play.position_assignments) {
       // Flatten the nested structure (DL, LB, DB)
       ['DL', 'LB', 'DB'].forEach(group => {
         if (play.position_assignments[group]) {
           Object.entries(play.position_assignments[group]).forEach(([pos, assign]: [string, any]) => {
             routes.push({
               receiver: pos,
               route: String(assign).split('-')[0].trim(), // Take the main action
               note: String(assign) // Keep full text as note
             });
           });
         }
       });
    }

    // Extract Reads
    const reads: string[] = [];
    if (play.reads) {
      if (play.reads.pre_snap) reads.push(**Pre-Snap:** ${play.reads.pre_snap});
      if (play.reads.primary) reads.push(**1️⃣ Primary:** ${play.reads.primary});
      if (play.reads.secondary) reads.push(**2️⃣ Secondary:** ${play.reads.secondary});
      if (play.reads.checkdown) reads.push(**3️⃣ Checkdown:** ${play.reads.checkdown});
      if (play.reads.hot_route && !play.reads.hot_route.includes('null')) {
         reads.push(**🚨 Hot:** ${play.reads.hot_route});
      }
    }

    // Build Usage Notes
    let notes = '';
    if (play.strategy?.beats) notes += **Beats:** ${play.strategy.beats}\n;
    if (play.coaching_points?.key_teaching) notes += **Coach's Key:** ${play.coaching_points.key_teaching};
    if (play.strategic_intel?.primary_weakness) notes += **Weakness:** ${play.strategic_intel.primary_weakness};

    schemes.push({
      name: internalId,
      display_name: displayName,
      system: system,
      formation_family: play.formation || "Unknown",
      routes: routes,
      reads: reads,
      usage_notes: notes,
      image_file: play.file_name || undefined
    });
  }
  return schemes;
}

import * as fs from 'fs';
import * as path from 'path';
import { ChatInputCommandInteraction, AutocompleteInteraction, EmbedBuilder, AttachmentBuilder } from 'discord.js';
import { getSchemeByName, searchSchemes } from '../knowledge/loader';
import { BRAND_CONFIGS } from '../knowledge/types';

export async function handleSchemeCommand(interaction: ChatInputCommandInteraction): Promise<void> {
  const schemeName = interaction.options.getString('name', true);
  const scheme = getSchemeByName(schemeName);
  
  if (!scheme) {
    await interaction.reply({ content: '❌ Scheme not found.', ephemeral: true });
    return;
  }
  
  const brand = BRAND_CONFIGS[scheme.system];
  
  // Format Routes List
  const routeText = scheme.routes.length > 0
    ? scheme.routes.slice(0, 8).map(r => • **${r.receiver}:** ${r.route}).join('\n')
    : 'No route data available.';

  // Format Reads List
  const readText = scheme.reads.length > 0
    ? scheme.reads.join('\n')
    : 'No read progression available.';

  const embed = new EmbedBuilder()
    .setColor(brand.color)
    .setTitle(${brand.emoji} ${scheme.display_name})
    .setDescription(**Formation:** ${scheme.formation_family})
    .addFields(
      { name: '📋 Assignments', value: routeText, inline: false },
      { name: '🧠 Reads / Keys', value: readText, inline: false },
      { name: '💡 Strategy', value: scheme.usage_notes || 'No specific notes.', inline: false }
    )
    .setFooter({ text: brand.footer })
    .setTimestamp();

  // Image Handling
  const files: AttachmentBuilder[] = [];
  
  if (scheme.image_file) {
    const imagePath = path.join(process.cwd(), 'assets', 'play_art', scheme.image_file);
    
    if (fs.existsSync(imagePath)) {
      const attachment = new AttachmentBuilder(imagePath, { name: 'play_art.png' });
      files.push(attachment);
      embed.setImage('attachment://play_art.png');
    }
  }

  await interaction.reply({ embeds: [embed], files: files });
}

export async function handleSchemeAutocomplete(interaction: AutocompleteInteraction): Promise<void> {
  const focusedValue = interaction.options.getFocused();
  const results = searchSchemes(focusedValue);
  await interaction.respond(results);
}
